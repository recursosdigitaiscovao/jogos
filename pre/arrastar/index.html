let nivelAtual = 1; let categoriaAtiva = null; let tempoSegundos = 0; let cronometro; let acertosNivel = 0; let pontosTotais = 0; // Inicializa o menu respeitando JOGO_CONFIG window.onload = function() { const container = document.getElementById('container-categorias'); for (const key in JOGO_CONFIG.categorias) { const cat = JOGO_CONFIG.categorias[key]; const btn = document.createElement('div'); btn.className = 'btn-categoria'; btn.innerText = cat.nome; btn.onclick = () => carregarCategoria(key); container.appendChild(btn); } }; function carregarCategoria(key) { categoriaAtiva = JOGO_CONFIG.categorias[key]; document.getElementById('menu-inicial').classList.add('hidden'); document.getElementById('tela-jogo').classList.remove('hidden'); iniciarCronometro(); montarFase(); } function iniciarCronometro() { cronometro = setInterval(() => { tempoSegundos++; const m = Math.floor(tempoSegundos / 60).toString().padStart(2, '0'); const s = (tempoSegundos % 60).toString().padStart(2, '0'); document.getElementById('display-tempo').innerText = `${m}:${s}`; }, 1000); } function montarFase() { acertosNivel = 0; const qtdNecessaria = (nivelAtual === 1) ? 6 : 8; document.getElementById('display-nivel').innerText = `Nível ${nivelAtual}`; // 1. Selecionar itens aleatórios da lista da categoria let itensSorteados = [...categoriaAtiva.lista] .sort(() => Math.random() - 0.5) .slice(0, qtdNecessaria); const containerObjetos = document.getElementById('col-objetos'); const containerSombras = document.getElementById('col-sombras'); containerObjetos.innerHTML = ''; containerSombras.innerHTML = ''; // 2. Criar peças coloridas (para arrastar) let ordemObjetos = [...itensSorteados].sort(() => Math.random() - 0.5); ordemObjetos.forEach(nomeImg => { const img = document.createElement('img'); img.src = categoriaAtiva.caminho + nomeImg; img.className = 'peca-original'; img.id = 'drag-' + nomeImg; img.draggable = true; img.ondragstart = (e) => e.dataTransfer.setData("text", nomeImg); containerObjetos.appendChild(img); }); // 3. Criar alvos (sombras) let ordemSombras = [...itensSorteados].sort(() => Math.random() - 0.5); ordemSombras.forEach(nomeImg => { const divAlvo = document.createElement('div'); divAlvo.className = 'alvo-sombra'; divAlvo.dataset.id = nomeImg; // Criamos a sombra via CSS filter usando a mesma imagem divAlvo.innerHTML = ``; divAlvo.ondragover = (e) => e.preventDefault(); divAlvo.ondrop = function(e) { e.preventDefault(); const data = e.dataTransfer.getData("text"); if (data === divAlvo.dataset.id) { divAlvo.classList.add('drop-sucesso'); document.getElementById('drag-' + data).style.visibility = 'hidden'; pontosTotais += 10; acertosNivel++; verificarFimDeNivel(qtdNecessaria); } }; containerSombras.appendChild(divAlvo); }); } function verificarFimDeNivel(total) { if (acertosNivel === total) { if (nivelAtual === 1) { setTimeout(() => { alert("Nível 1 Concluído! Prepare-se para o Nível 2."); nivelAtual = 2; montarFase(); }, 500); } else { finalizarJogo(); } } } function finalizarJogo() { clearInterval(cronometro); const rel = JOGO_CONFIG.relatorio; document.getElementById('modal-relatorio').classList.remove('hidden'); document.getElementById('rel-titulo').innerText = rel.titulo; document.getElementById('rel-pontos').innerText = `${rel.pontosTotal} ${pontosTotais}`; document.getElementById('rel-tempo').innerText = `${rel.tempoTotal} ${document.getElementById('display-tempo').innerText}`; }
