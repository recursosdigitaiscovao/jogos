function loadRound() { const cat = JOGO_CONFIG.categorias[currentCatKey]; const item = cat.itens[currentQuestion]; const footer = document.getElementById('game-interaction-area'); const header = document.getElementById('word-header'); // ALTERAÇÃO: Se for ordenar, esconde o nome. Se for contar, mostra o nome. if(cat.tipo === 'ordenar') { header.style.display = 'none'; } else { header.style.display = 'block'; header.innerText = item.nome; } document.getElementById('current-img').src = JOGO_CONFIG.caminhoImg + item.img; footer.innerHTML = ''; userSyllables = []; if(cat.tipo === 'contar') { const grid = document.createElement('div'); grid.className = 'contar-container'; [1,2,3,4].forEach(n => { const b = document.createElement('div'); b.className = 'btn-num'; b.innerText = n; b.onclick = () => checkContar(n); grid.appendChild(b); }); footer.appendChild(grid); } else { const slotsArea = document.createElement('div'); slotsArea.className = 'slots-area'; const syllaArray = item.divisao.split('-'); syllaArray.forEach(() => { const slot = document.createElement('div'); slot.className = 'slot'; slotsArea.appendChild(slot); }); footer.appendChild(slotsArea); const optsArea = document.createElement('div'); optsArea.className = 'syllable-options'; const shuffled = [...syllaArray].sort(() => Math.random() - 0.5); shuffled.forEach(syl => { const btn = document.createElement('div'); btn.className = 'syllable-btn'; btn.innerText = syl; btn.onclick = () => { if(userSyllables.length < syllaArray.length && !btn.classList.contains('used')) { btn.classList.add('used'); const slots = document.querySelectorAll('.slot'); slots[userSyllables.length].innerText = syl; slots[userSyllables.length].classList.add('filled'); userSyllables.push(syl); if(userSyllables.length === syllaArray.length) checkOrder(syllaArray); } }; optsArea.appendChild(btn); }); footer.appendChild(optsArea); } }
